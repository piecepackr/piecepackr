Usage: Program to collect piecepacks


Options:
	-h, --help
		Show this help message and exit

	--author=AUTHOR
		Pdf author (default "")

	--filename=FILENAME
		Filename prefix (default "piecepack_collection")

	--keywords=KEYWORDS
		Pdf keywords (default "piecepack")

	--subject=SUBJECT
		Pdf subject (default "")

	--title=TITLE
		Pdf title (default "Piecepack collection")

	--decks=DECKS
		Comma separated list of decks to collect (default collects all of them)

	--pdf_deck_dir=PDF_DECK_DIR
		(default "pdf/decks")

	--pdf_collection_dir=PDF_COLLECTION_DIR
		(default "pdf/collections")

	--pdf_preview_dir=PDF_PREVIEW_DIR
		(default "pdf/previews")

	--svg_preview_dir=SVG_PREVIEW_DIR
		(default "svg/previews")

	--paper=PAPER
		Pdf paper size, either "letter" or "A4" (default "letter")

	--font=FONT
		Default font family (default "sans")
