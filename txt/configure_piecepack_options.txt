Usage: Program to make piecepack configurations


Options:
	-h, --help
		Show this help message and exit

	--deck_title=DECK_TITLE
		(default "")

	--deck_filename=DECK_FILENAME
		Filename prefix (default "piecepack_deck")

	--rank_symbols=RANK_SYMBOLS
		(default "N,A,2,3,4,5")

	--suit_symbols=SUIT_SYMBOLS
		(default "♥,♠,♣,♦,☼")

	--dm_symbols=DM_SYMBOLS
		(default is to try to pick a reasonable directional mark symbol based on the component)

	--header_font=HEADER_FONT
		(default is to use the value of "font" option)

	--rank_symbols_font=RANK_SYMBOLS_FONT
		(default is to use the value of "font" option)

	--suit_symbols_font=SUIT_SYMBOLS_FONT
		(default is to use the value of "font" option)

	--dm_symbols_font=DM_SYMBOLS_FONT
		(default is to use the value of "font" option)

	--rank_symbols_scale=RANK_SYMBOLS_SCALE
		(default is not to adjust scale)

	--suit_symbols_scale=SUIT_SYMBOLS_SCALE
		(default is not to adjust scale)

	--dm_symbols_scale=DM_SYMBOLS_SCALE
		(default is not to adjust scale)

	--use_suit_as_ace
		(default "FALSE")

	--use_ace_as_ace
		Opposite of "use_suit_as_ace" option

	--pp_die_arrangement=PP_DIE_ARRANGEMENT
		Either "counter" or "opposites_sum_to_5" (default "counter")

	--shape=SHAPE
		either "rect", "circle", "halma", "kite", "star", or a number representing number of sides of a regular polygon (default is to choose reasonable shape based on component)'

	--shape_theta=SHAPE_THETA
		If shape is a number or "star" then angle of first vertex (in degrees) of polygon (default is "90")

	--checker_colors=CHECKER_COLORS
		(default is not to draw any checkers on tiles)

	--gridline_colors=GRIDLINE_COLORS
		(default is to draw gridlines on tile back using the "unsuit" suit color)

	--hexline_colors=HEXLINE_COLORS
		(default is not to draw any hexlines on tiles)

	--dm_theta=DM_THETA
		Angle (in degrees) of polar coordinates of direction mark (default 135 for tile faces and die faces and 90 for everything else)

	--dm_r=DM_R
		Radius from center (relative units) of polar coordinates of direction mark (default "sqrt(.25^2+.25^2)" if its shape is "rect" or "circle" otherwise "0.3")

	--ps_theta=PS_THETA
		Angle (in degrees) of polar coordinates of primary symbol (default 135 for tile faces and die faces and 90 for everything else)

	--ps_r=PS_R
		Radius from center (relative units) of polar coordinates of primary symbol (default "sqrt(.25^2+.25^2)" if its shape is "rect" or "circle" otherwise "0.3")

	--font=FONT
		Default font family (default "sans")
